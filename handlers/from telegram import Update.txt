from telegram import Update
from telegram.ext import ContextTypes
from services.referral_system import db

async def refstats(update:Update,context:ContextTypes.DEFAULT_TYPE):

    con = db()
    cur = con.cursor()

    cur.execute("SELECT COUNT(*) FROM referral_clicks")
    clicks = cur.fetchone()[0]

    cur.execute("SELECT COUNT(*) FROM referrals")
    total = cur.fetchone()[0]

    cur.execute("SELECT COUNT(*) FROM referrals WHERE status='pending'")
    pending = cur.fetchone()[0]

    cur.execute("SELECT COUNT(*) FROM referrals WHERE status='qualified'")
    approved = cur.fetchone()[0]

    text = f"""
📊 <b>Análise global de indicações</b>

👆 Cliques totais: {clicks}

📨 Registradas: {total}

⏳ Pendentes: {pending}

✅ Aprovadas: {approved}
"""

    await update.message.reply_text(text,parse_mode="HTML")
